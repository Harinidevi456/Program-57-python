import math
r = float(input(&quot;Input the radius of the circle: &quot;))
c = 2 * math.pi * r
area = math.pi * r * r
print(&quot;The circumference of the circle is: &quot;, c)
print(&quot;The area of the circle is: &quot;, area)   
output
The circumference of the circle is: 43.982
The area of the circle is: 153.93
